B2. SUPERUSER DASHBOARD — AGENT PERFORMANCE MATRIX (BubbleChart2D)
    Component: fulldashboard.jsx → BubbleChart2D (SVG)
--------------------------------------------------------------------------------

COMPONENT        : BubbleChart2D — Agent bubble chart
FEATURE          : SVG scatter — X=calls handled, Y=CSAT, size=workload%, color=risk
API ENDPOINT     : GET /api/superuser/realtime
TABLE            : agent_states (primary), cc_sessions (via CTE)
COLUMNS          :
  agent_states  — agent_identity, agent_name, department, status, last_heartbeat, sequence_number
  cc_sessions   — agent_id, status, created_at
SQL (CTE)        :
  WITH agg AS (
    SELECT agent_id,
           COUNT(*) FILTER (WHERE status='completed') AS calls_handled,
           COUNT(*) FILTER (WHERE status IN ('abandoned','transferred')) AS escalations
    FROM cc_sessions WHERE created_at > NOW() - INTERVAL '24 hours' AND agent_id != ''
    GROUP BY agent_id
  )
  SELECT a.agent_identity, a.agent_name, a.department, a.status,
         COALESCE(g.calls_handled, 0) AS calls_handled,
         COALESCE(g.escalations, 0) AS escalations,
         a.last_heartbeat
  FROM agent_states a LEFT JOIN agg g ON g.agent_id = a.agent_identity
  WHERE a.last_heartbeat > NOW() - INTERVAL '10 minutes'
POLLING          : 5s
STATUS           : [DB]
NOTE             : calls_handled is NOT a column in agent_states — computed via CTE

  SECTION D: COMPLETE DATABASE TABLE REFERENCE

TABLE: cc_sessions
  Purpose  : Every call — inbound, outbound, direct
  Columns  : id, session_id, room_id, caller_id (FK→cc_callers), department,
             status, call_duration, agent_id, created_at, ended_at,
             recording_url, recording_consent, wait_seconds, csat,
             sla_target_seconds, sla_breached
  Used by  : Almost every component
  IMPORTANT: calls_handled is NOT a column — always computed via COUNT(*) CTE

TABLE: cc_callers
  Purpose  : Caller / user identity directory
  Columns  : id, email, display_name, last_seen
  Used by  : Agent call list, ActiveCallsPanel, HistoryPanel, AgentDetail

TABLE: agent_states
  Purpose  : Agent presence, status, identity
  Columns  : agent_identity, agent_name, department, status, last_heartbeat,
             sequence_number, ignore_outbounds_until, ignore_reason
  Used by  : QueueMonitor, BubbleChart2D, Realtime endpoint, all agent heartbeat calls
  IMPORTANT: calls_handled NOT a column — computed from cc_sessions via CTE

TABLE: transcripts
  Purpose  : Per-utterance call dialogue (speaker + text)
  Columns  : id, call_id, session_id, speaker, text, sentiment, created_at
  Used by  : TranscriptModal in ActiveCallsPanel, HistoryPanel, Admindashboard

TABLE: outbound_queue
  Purpose  : Scheduled callbacks and outbound campaign jobs
  Columns  : id, user_email, department, status, assigned_agent, attempts,
             created_at, last_attempt, scheduler_email_sent
  Used by  : Scheduling page, BroadcastPanel (email campaign history),
             OutboundCallNotification, callcenter/api outbound flow

TABLE: admin_config
  Purpose  : Key-value system configuration store
  Columns  : key (TEXT), value (TEXT/JSON)
  Key rows : business_hours, holiday_active, holiday_message,
             smtp_host, smtp_port, smtp_user, smtp_pass, smtp_from,
             max_wait_seconds, work_start, work_end, timezone
  Used by  : BusinessHours, Settings, BroadcastPanel (SMTP status)

TABLE: call_summaries
  Purpose  : AI-generated post-call summaries and sentiment
  Columns  : id, call_id, session_id, summary_text, sentiment, created_at
  Used by  : End-call flow in home_routes; referenced indirectly by dashboard

TABLE: users
  Purpose  : User accounts (login, roles, credits)
  Columns  : id, name, email, role, is_active, phone_number, created_at, credits
  Used by  : AdminSettings only (not directly used by agent or superuser pages)
